# regression
import statsmodels.api as sm
def regress(data,yvar,xvars):
	Y=data[yvar]
	X=data[xvars]
	X['intercept']=1.
	result=sm.OLS(Y,X).fit()
	return result.params
by_year.apply(regress,'AAPL',['SPX'])
